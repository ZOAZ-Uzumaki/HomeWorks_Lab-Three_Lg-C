/* 21.   Дано ціле число N. Знайти найменше ціле додатне число K,
 *       квадрат якого перевершує N: K2 > N.
 *       Функцію отримання квадратного кореня не використовувати.
 * */


int n = 16;
int k = 1;

while (k * k <= n)
{
    k++;
}

Console.WriteLine("Найменше додатнє число K -  " + k);
